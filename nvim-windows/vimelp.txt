-- Useful stuff
g + r + a -- look at code suggestion
g + r + r -- goto references
g + r + n -- rename across files
g + r + d -- goto definition
Ctrl + o -- cursor go back to where it was
Ctrl + i -- cursor go forward
:wa -- save all modified buffers
d + i + ' or " -- delete stuff inside quotes

Ctrl + b -- neotree toggle
Ctrl + n -- create new split terminal
Ctrl + p -- create new split terminal in current buffer
Ctrl + q -- quick quit
Space + c -- cmake build and run
